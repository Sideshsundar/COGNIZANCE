Function Main
    Declare Integer F1, F2, FN, N, X
    
    Output "Enter the number of terms:"
    Input N
    Assign F1 = 0
    Assign F2 = 1
    If N < 1
    End
    Output F1
    For X = 0 to N
        Output F2
        Assign FN = F1 + F2
        Assign F1 = F2
        Assign F2 = FN
        Assign X = X + 1
    End
    Output "The value of N-th term in fibonacci series is :"
    Output FN
End
