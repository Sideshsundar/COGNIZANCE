Function Main
    Declare Integer N, D, R, I
    
    Assign D = 3
    Output "Enter value for N:"
    Input N
    
    ... remainder = divident - divisor * (divident // divisor) 
    ... D = divisor 
    ... N = divident 
    ... R = remainder
    For I = 0 to N
        If N > 0
            Assign R = N-D
            Assign N = R
        End
    End
    If R == 0
        Output "It is divisible by 3 . Hence True"
    False:
        Output "It is not divisible by 3 . Hence False"
    End
End
